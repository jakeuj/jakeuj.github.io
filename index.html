tasks.json

{
    // See https://go.microsoft.com/fwlink/?LinkId=733558 
    // for the documentation about the tasks.json format
    "version": "2.0.0",
    "tasks": [
        {
            "label": "Hexo Deploy",
            "type": "shell",
            "dependsOn": "Hexo Clean",
            "command": "hexo g --d",
            "group": {
                "kind": "build",
                "isDefault": true
            }
        },
        {
            "label": "Hexo Clean",
            "type": "shell",
            "command": "hexo clean",
            "group": "build"
        },
        {
            "label": "Hexo Server",
            "type": "shell",
            "command": "hexo server",       
            "group":"test",
        },
        {
            "label": "Hexo Preview",
            "type": "process",
            "dependsOn": "Hexo Server",
            "args": ["http://localhost:4000/"],

                "command": "C:/Program Files (x86)/Google/Chrome/Application/chrome.exe"
            },
            "group": {
                "kind": "test",
                "isDefault": true
            }
        }
    ]
}

<section id="comment">
  <h1 class="title"><%= __('comment') %></h1>
